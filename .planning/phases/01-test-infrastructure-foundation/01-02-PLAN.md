Create worker-lifecycle.ts with WorkerLifecycle class:
- `waitForWorkerReady(workerType: 'image' | 'audio' | 'document' | 'spreadsheet' | 'archive', timeout?: number): Promise<void>`
  - Uses `Promise.all` to wait for both:
    1. Worker script response: `page.waitForResponse(r => r.url().includes(\`${workerType}-worker.js\`))`
    2. Worker initialization: `page.waitForFunction()` checking window.__workerManager or similar
  - Default timeout 5000ms, configurable
- `terminateAll(): Promise<void>` - Calls into page context to terminate worker manager
- Tracks active workers for cleanup

Create timeout-config.ts:
- `calculateTimeout(fileSizeMB: number, complexity: 'simple' | 'medium' | 'complex'): number`
  - Base: 30000ms
  - Per MB: 2000ms
  - Complexity multipliers: simple=1, medium=2, complex=4
  - Returns timeout in milliseconds
- `applyTimeout(testInfo: TestInfo, fileSizeMB: number, complexity: 'simple' | 'medium' | 'complex'): void`
  - Calls `testInfo.setTimeout(calculatedTimeout)`
- Export type `TimeoutConfig = { base: number, perMB: number, multipliers: Record<string, number> }`

Create index.ts that exports extended test with all fixtures:

```typescript
import { test as base, expect } from '@playwright/test';
import { FileHelper } from './file-helpers';
import { DownloadHelper } from './download-helpers';
import { WorkerLifecycle } from './worker-lifecycle';

type ConversionFixtures = {
  fileHelper: FileHelper;
  downloadHelper: DownloadHelper;
  workerLifecycle: WorkerLifecycle;
};

export const test = base.extend<ConversionFixtures>({
  fileHelper: async ({ page }, use) => {
    const helper = new FileHelper(page);
    await use(helper);
  },

  downloadHelper: async ({ page }, use) => {
    const helper = new DownloadHelper(page);
    await use(helper);
    await helper.cleanup();
  },

  workerLifecycle: async ({ page }, use) => {
    const helper = new WorkerLifecycle(page);
    await use(helper);
    await helper.terminateAll();
  }
});

export { expect };
export { calculateTimeout, applyTimeout } from './timeout-config';
export type { TimeoutConfig } from './timeout-config';
```

This allows tests to import:
```typescript
import { test, expect, calculateTimeout } from './fixtures';

test('my test', async ({ page, fileHelper, downloadHelper, workerLifecycle }) => {
  // All fixtures available
});
```
