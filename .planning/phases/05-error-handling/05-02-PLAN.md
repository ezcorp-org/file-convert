```typescript
import { test, expect } from '../../fixtures';
import { ImageFactory, AudioFactory } from '../../fixtures/factories';

test.describe('ERROR-05: Extension Spoofing Detection', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('/convert');
    await page.waitForLoadState('networkidle');
  });

  test('detects JPEG file with PNG extension', async ({ page }) => {
    // Create valid JPEG using ImageFactory
    const jpegBuffer = await ImageFactory.createJPEG({ width: 100, height: 100 });

    // Upload with .png extension (spoofed)
    await page.locator('input[type="file"]').setInputFiles({
      name: 'spoofed.png',
      mimeType: 'image/png',
      buffer: jpegBuffer
    });

    // Check for warning/error about format mismatch
    // Per CONTEXT.md: "Warn but allow" - look for warning notification
    const notification = page.locator('.notification, [class*="notification"]');
    await expect(notification.first()).toBeVisible({ timeout: 5000 });

    // Verify message mentions format detection
    const text = await notification.first().textContent();
    console.log('Spoofing notification:', text);
    // Should contain something like "appears to be JPEG" or "format mismatch"
    expect(text?.toLowerCase()).toMatch(/jpeg|mismatch|appears|actual/i);
  });

  test('detects PNG file with JPEG extension', async ({ page }) => {
    const pngBuffer = await ImageFactory.createPNG({ width: 100, height: 100 });

    await page.locator('input[type="file"]').setInputFiles({
      name: 'spoofed.jpg',
      mimeType: 'image/jpeg',
      buffer: pngBuffer
    });

    const notification = page.locator('.notification, [class*="notification"]');
    await expect(notification.first()).toBeVisible({ timeout: 5000 });

    const text = await notification.first().textContent();
    console.log('Spoofing notification:', text);
    expect(text?.toLowerCase()).toMatch(/png|mismatch|appears|actual/i);
  });

  test('detects WAV file with MP3 extension', async ({ page }) => {
    const wavBuffer = await AudioFactory.createWAV({
      duration: 0.5,
      sampleRate: 44100
    });

    await page.locator('input[type="file"]').setInputFiles({
      name: 'spoofed.mp3',
      mimeType: 'audio/mpeg',
      buffer: wavBuffer
    });

    const notification = page.locator('.notification, [class*="notification"]');
    await expect(notification.first()).toBeVisible({ timeout: 5000 });

    const text = await notification.first().textContent();
    console.log('Spoofing notification:', text);
    expect(text?.toLowerCase()).toMatch(/wav|mismatch|appears|actual/i);
  });

  test('allows correct file without warning', async ({ page }) => {
    // Create PNG and upload as PNG - should not show warning
    const pngBuffer = await ImageFactory.createPNG({ width: 100, height: 100 });

    await page.locator('input[type="file"]').setInputFiles({
      name: 'correct.png',
      mimeType: 'image/png',
      buffer: pngBuffer
    });

    // Wait for file to appear in list (success case)
    const fileItem = page.locator('.file-item, [class*="file-item"], .files-list');
    await expect(fileItem.first()).toBeVisible({ timeout: 5000 });

    // Should NOT have error/warning notification for correct file
    const errorNotification = page.locator('.notification--error, .notification--warning');
    // Use count() instead of toBeHidden to avoid flaky behavior
    const errorCount = await errorNotification.count();
    // Might have 0 or error could be from previous test - check if text mentions our file
    if (errorCount > 0) {
      const text = await errorNotification.first().textContent();
      expect(text?.toLowerCase()).not.toContain('correct.png');
    }
  });
});
```
