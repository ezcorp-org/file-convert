Create E2E test suite for worker crash recovery (ERROR-06).

**Test strategy:**
Worker crashes are hard to simulate directly in E2E tests. Instead, test behaviors that would result from worker failures:

1. **Severely corrupted file that causes worker to error** - File that passes initial validation but crashes during processing
2. **UI responsiveness after failure** - Verify UI remains interactive
3. **Error message visibility** - Verify user sees failure notification
4. **Retry behavior** - Check for retry indicators if implemented

**Test structure:**
```typescript
import { test, expect } from '../../fixtures';
import { CorruptedFileFactory } from '../../fixtures/factories';
import { ImageFactory } from '../../fixtures/factories';

test.describe('ERROR-06: Worker Crash Recovery', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('/convert');
    await page.waitForLoadState('networkidle');
  });

  test('UI remains responsive after conversion failure', async ({ page }) => {
    // Upload a file that will fail during conversion
    // (truncated file with valid header - passes upload, fails processing)
    const truncatedPng = CorruptedFileFactory.createTruncatedFile('png', 100);

    await page.locator('input[type="file"]').setInputFiles({
      name: 'truncated.png',
      mimeType: 'image/png',
      buffer: truncatedPng
    });

    // Wait for file to appear
    const fileItem = page.locator('.file-item, [class*="file"]');
    await expect(fileItem.first()).toBeVisible({ timeout: 5000 });

    // Select output format and convert
    const formatButton = page.locator('button:has-text("JPEG"), button:has-text("JPG")').first();
    if (await formatButton.isVisible()) {
      await formatButton.click();
    }

    const convertButton = page.locator('button:has-text("Convert")').first();
    if (await convertButton.isVisible()) {
      await convertButton.click();
    }

    // Wait for failure notification (conversion should fail)
    const notification = page.locator('.notification, [class*="notification"]');
    await expect(notification.first()).toBeVisible({ timeout: 15000 });

    // UI RESPONSIVENESS CHECK: Can still interact with page
    // Try uploading another file - proves UI isn't frozen
    const validPng = await ImageFactory.createPNG({ width: 50, height: 50 });

    await page.locator('input[type="file"]').setInputFiles({
      name: 'valid-after-failure.png',
      mimeType: 'image/png',
      buffer: validPng
    });

    // Should be able to see the new file (proves UI responsive)
    await expect(page.locator('text=valid-after-failure')).toBeVisible({ timeout: 5000 });
  });

  test('shows error notification for failed conversion', async ({ page }) => {
    // Create file that will fail conversion
    const badFile = CorruptedFileFactory.createBadHeaderFile('png');

    await page.locator('input[type="file"]').setInputFiles({
      name: 'will-fail.png',
      mimeType: 'image/png',
      buffer: badFile
    });

    // Try to convert (may fail at upload or conversion)
    const convertButton = page.locator('button:has-text("Convert")');
    if (await convertButton.first().isVisible({ timeout: 3000 }).catch(() => false)) {
      await convertButton.first().click();
    }

    // Should see some error notification
    const errorNotification = page.locator('.notification--error, [class*="error"]');
    await expect(errorNotification.first()).toBeVisible({ timeout: 10000 });

    // Error should have meaningful message (not just "Error")
    const errorText = await errorNotification.first().textContent();
    console.log('Worker failure error:', errorText);
    expect(errorText?.length).toBeGreaterThan(5); // Not empty/minimal
  });

  test('can successfully convert after previous failure', async ({ page }) => {
    // First, trigger a failure
    const badFile = CorruptedFileFactory.createTruncatedFile('png', 50);

    await page.locator('input[type="file"]').setInputFiles({
      name: 'fail-first.png',
      mimeType: 'image/png',
      buffer: badFile
    });

    // Wait for upload handling
    await page.waitForTimeout(1000);

    // Clear and try with valid file
    // Look for clear/remove button or reload page
    const clearButton = page.locator('button:has-text("Clear"), button:has-text("Remove"), [class*="clear"]');
    if (await clearButton.first().isVisible().catch(() => false)) {
      await clearButton.first().click();
    } else {
      // No clear button - reload to fresh state
      await page.goto('/convert');
      await page.waitForLoadState('networkidle');
    }

    // Now upload valid file
    const validPng = await ImageFactory.createPNG({ width: 100, height: 100 });

    await page.locator('input[type="file"]').setInputFiles({
      name: 'valid-second.png',
      mimeType: 'image/png',
      buffer: validPng
    });

    // Select JPEG output
    const jpegButton = page.locator('button:has-text("JPEG"), button:has-text("JPG")').first();
    await expect(jpegButton).toBeVisible({ timeout: 5000 });
    await jpegButton.click();

    // Convert
    const convertButton = page.locator('button:has-text("Convert")').first();
    await convertButton.click();

    // Should succeed (worker recovered/restarted)
    // Wait for download button or success indicator
    const successIndicator = page.locator(
      'button:has-text("Download"), .download-button, [class*="success"], [class*="complete"]'
    );
    await expect(successIndicator.first()).toBeVisible({ timeout: 30000 });
  });

  test.skip('shows retry indicator when worker retries', async ({ page }) => {
    // SKIP REASON: Testing retry behavior requires either:
    // 1. Mocking worker to fail once then succeed (complex in E2E)
    // 2. A file that consistently fails first attempt but succeeds on retry (unreliable)
    //
    // Per CONTEXT.md, retry behavior should show "Retrying (attempt 2/2)..."
    // This test documents expected behavior but can't reliably test it E2E.
    //
    // TO TEST: Use unit tests on worker-manager.ts with mocked workers
    // or add a test flag that forces retry behavior for testing purposes.
  });
});
```

**Implementation notes:**
- Worker crashes are hard to simulate - focus on observable behavior
- Use truncated/corrupted files to trigger conversion failures
- Key assertion: UI remains responsive (can upload new file after failure)
- Key assertion: Error notification is visible with meaningful message
- Key assertion: Subsequent valid conversion still works
- Skip tests that require mock injection (document why)
