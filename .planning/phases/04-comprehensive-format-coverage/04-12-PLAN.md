```typescript
test('WAV to FLAC to WAV is truly lossless', async ({
  page,
  fileHelper,
  downloadHelper

    duration: 1,
    sampleRate: 44100,
    channels: 2,
    bitDepth: 16,
    frequency: 440
  });

  const originalSamples = AudioFactory.getSampleCount(originalWav);

  await page.goto('/convert');
  await page.waitForLoadState('networkidle');

  await flacOption.click();

  const { buffer: flacBuffer } = await downloadHelper.validateDownload('.download-btn', 'flac');

  // Verify FLAC magic bytes
  expect(flacBuffer[0]).toBe(0x66);
  expect(flacBuffer[1]).toBe(0x4C);

  // Step 2: Convert FLAC -> WAV
  await page.goto('/convert');
  await page.waitForLoadState('networkidle');

  const flacFileData = fileHelper.createFileData(flacBuffer, 'test.flac', 'audio/flac');
  await fileHelper.uploadFile(flacFileData);
  await expect(page.locator('.file-item')).toContainText('test.flac');

  const wavOption = page.locator('.format-option').filter({ hasText: /WAV/i });
  await wavOption.click();
  await page.locator('.convert-btn').first().click();
  await expect(page.locator('.download-btn').first()).toBeVisible({ timeout: 45000 });

  const { buffer: finalWavBuffer } = await downloadHelper.validateDownload('.download-btn', 'wav');

  // Step 3: Validate lossless preservation
  const finalDuration = AudioFactory.getDuration(finalWavBuffer);
  const finalSamples = AudioFactory.getSampleCount(finalWavBuffer);

  // Duration should match within 0.01s tolerance (header differences)
  expect(Math.abs(originalDuration - finalDuration)).toBeLessThan(0.01);

  // Sample count MUST match exactly for lossless
  expect(finalSamples).toBe(originalSamples);

  console.log('Lossless round-trip:', {
    originalSamples,
    finalSamples,
    originalDuration,
    finalDuration,
    samplesMatch: finalSamples === originalSamples
  });
});
```

```typescript
test('MP3 conversion maintains reasonable quality', async ({
  page,
  fileHelper,
  downloadHelper
}) => {
  const originalWav = AudioFactory.createWAV({
    duration: 1,
    sampleRate: 44100,
    channels: 2,
    bitDepth: 16,
    frequency: 440
  });

  const originalDuration = AudioFactory.getDuration(originalWav);

  // Convert to MP3
  await page.goto('/convert');
  await page.waitForLoadState('networkidle');

  const fileData = fileHelper.createFileData(originalWav, 'test.wav', 'audio/wav');
  await fileHelper.uploadFile(fileData);
  await expect(page.locator('.file-item')).toContainText('test.wav');

  const mp3Option = page.locator('.format-option').filter({ hasText: /MP3/i });
  await mp3Option.click();
  await page.locator('.convert-btn').first().click();
  await expect(page.locator('.download-btn').first()).toBeVisible({ timeout: 45000 });

  const { buffer: mp3Buffer } = await downloadHelper.validateDownload('.download-btn', 'mp3');

  // Validate audio structure using StructuralValidator
  const audioValidation = await StructuralValidator.validateAudio(mp3Buffer);
  expect(audioValidation.valid).toBe(true);
  expect(audioValidation.format).toBeTruthy();

  // Validate duration preserved (within 0.1s tolerance for lossy codec)
  if (audioValidation.metadata?.duration) {
    const durationDiff = Math.abs(audioValidation.metadata.duration - originalDuration);
    expect(durationDiff).toBeLessThan(0.1);
  }

  // Validate bitrate is reasonable (>64 kbps for standard quality)
  if (audioValidation.metadata?.bitrate) {
    expect(audioValidation.metadata.bitrate).toBeGreaterThan(64000);
  }

  // Validate lossy compression reduced file size
  expect(mp3Buffer.length).toBeLessThan(originalWav.length);

  console.log('MP3 quality validation:', {
    originalDuration,
    mp3Duration: audioValidation.metadata?.duration,
    bitrate: audioValidation.metadata?.bitrate,
    originalSize: originalWav.length,
    mp3Size: mp3Buffer.length,
    compressionRatio: (originalWav.length / mp3Buffer.length).toFixed(2)
  });
});
```

Key validations:
- StructuralValidator.validateAudio() extracts metadata.duration and metadata.bitrate
- Duration within 0.1s of original (lossy codec tolerance)
- Bitrate > 64000 bps (reasonable quality)
- File size < WAV size (proves lossy compression)
