1. Create a Node.js script to generate the image:
   ```javascript
   // apps/frontend/tests/testAssets/generate-exif-image.js
   const sharp = require('sharp');
   const fs = require('fs');
   const path = require('path');

   async function generateExifImage() {
     // Create a 200x200 image with visible content (for visual debugging)
     const image = sharp({
       create: {
         width: 200,
         height: 200,
         channels: 3,
         background: { r: 100, g: 149, b: 237 }  // Cornflower blue
       }
     });

     // Add comprehensive EXIF metadata
     const buffer = await image
       .withMetadata({
         exif: {
           IFD0: {
             Make: 'TestCamera',
             Model: 'EXIF-Test-Model',
             Software: 'FileConvert-Test',
             DateTime: '2026:01:24 12:00:00',
             Artist: 'Test Author',
             Copyright: 'Test Copyright 2026',
             ImageDescription: 'Test image for metadata preservation'
           },
           IFD1: {},
           EXIF: {
             DateTimeOriginal: '2026:01:24 12:00:00',
             DateTimeDigitized: '2026:01:24 12:00:00',
             ExposureTime: '1/60',
             FNumber: '2.8',
             ISOSpeedRatings: 400,
             FocalLength: '50',
             ColorSpace: 1,
             PixelXDimension: 200,
             PixelYDimension: 200
           },
           GPS: {
             GPSLatitudeRef: 'N',
             GPSLatitude: [[37, 1], [46, 1], [30, 1]],
             GPSLongitudeRef: 'W',
             GPSLongitude: [[122, 1], [25, 1], [10, 1]]
           }
         }
       })
       .jpeg({ quality: 95 })
       .toBuffer();

     // Write to testAssets
     const outputPath = path.join(__dirname, 'images', 'sample-with-exif.jpg');
     fs.mkdirSync(path.dirname(outputPath), { recursive: true });
     fs.writeFileSync(outputPath, buffer);

     console.log(`Created ${outputPath} (${buffer.length} bytes)`);

     // Verify EXIF is readable
     const ExifReader = require('exifreader');
     const tags = await ExifReader.load(buffer);
     console.log('EXIF verification:');
     console.log('  Make:', tags.Make?.description);
     console.log('  Model:', tags.Model?.description);
     console.log('  DateTime:', tags.DateTime?.description);
   }

   generateExifImage().catch(console.error);
   ```

2. Run the script:
   ```bash

   ```
