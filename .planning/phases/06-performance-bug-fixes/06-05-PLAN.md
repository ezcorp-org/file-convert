import { test, expect } from '@playwright/test';
import { FileHelper, ImageFactory, AudioFactory, ArchiveFactory } from '../../fixtures';

const LARGE_IMAGE_SIZE_MB = 10;
const LARGE_AUDIO_SIZE_MB = 25;
const LARGE_ARCHIVE_SIZE_MB = 50;

test.describe('Large File Handling', () => {
  test.describe.configure({ timeout: 300000 }); // 5 minute timeout for large files

  test.describe('Large Image Files', () => {
    test('converts 10MB PNG to JPEG without memory error', async ({ page }) => {
      await page.goto('/convert');
      await page.waitForLoadState('networkidle');

      // Generate large PNG (10MB = ~1800x1800 RGBA uncompressed, but PNG compresses)
      // For 10MB file, need larger dimensions
      const imageBuffer = await ImageFactory.create({
        width: 3000,
        height: 3000,
        format: 'png'
      });

      // Verify size is approximately 10MB (PNG compression varies)
      console.log(`Generated image size: ${(imageBuffer.length / 1024 / 1024).toFixed(2)}MB`);

      // Upload and convert
      const fileHelper = new FileHelper(page);
      await fileHelper.uploadBuffer(imageBuffer, 'large-image.png', 'image/png');

      // Wait for file to appear in list
      await expect(page.locator('.file-item')).toBeVisible({ timeout: 30000 });

      // Select JPEG output
      await page.locator('.format-selector').click();
      await page.locator('text=/JPEG/i').click();

      // Start conversion
      await page.locator('button:has-text("Convert")').click();

      // Wait for completion (generous timeout for large file)
      await expect(page.locator('.download-button, .success-indicator, [data-status="complete"]'))
        .toBeVisible({ timeout: 180000 });

      // Verify no memory error message
      const errorVisible = await page.locator('text=/memory|out of memory|too large/i').isVisible();
      expect(errorVisible).toBe(false);
    });

    test.skip('shows clear error for oversized file (>100MB)', async ({ page }) => {
      // Skip in CI - would generate very large file
      // This test validates graceful degradation
    });
  });

  test.describe('Large Audio Files', () => {
    test('converts 25MB WAV to MP3 without memory error', async ({ page }) => {
      await page.goto('/convert');
      await page.waitForLoadState('networkidle');

      // 25MB WAV = ~143 seconds at 44.1kHz, 16-bit, stereo
      // 25MB / (44100 * 2 * 2) = ~142 seconds
      const audioBuffer = await AudioFactory.create({
        duration: 140,  // seconds
        sampleRate: 44100,
        channels: 2
      });

      console.log(`Generated audio size: ${(audioBuffer.length / 1024 / 1024).toFixed(2)}MB`);

      const fileHelper = new FileHelper(page);
      await fileHelper.uploadBuffer(audioBuffer, 'large-audio.wav', 'audio/wav');

      await expect(page.locator('.file-item')).toBeVisible({ timeout: 30000 });

      // Select MP3 output
      await page.locator('.format-selector').click();
      await page.locator('text=/MP3/i').click();

      await page.locator('button:has-text("Convert")').click();

      // Longer timeout for audio conversion
      await expect(page.locator('.download-button, .success-indicator, [data-status="complete"]'))
        .toBeVisible({ timeout: 240000 });

      const errorVisible = await page.locator('text=/memory|out of memory|too large/i').isVisible();
      expect(errorVisible).toBe(false);
    });
  });

  test.describe('Large Archive Files', () => {
    test('converts 50MB ZIP to TAR without memory error', async ({ page }) => {
      await page.goto('/convert');
      await page.waitForLoadState('networkidle');

      // Generate 50MB archive with multiple files
      const archiveBuffer = await ArchiveFactory.create({

          { name: 'large-file-3.txt', content: 'C'.repeat(10 * 1024 * 1024) }   // 10MB

      await fileHelper.uploadBuffer(archiveBuffer, 'large-archive.zip', 'application/zip');

      await expect(page.locator('.download-button, .success-indicator, [data-status="complete"]'))
        .toBeVisible({ timeout: 300000 });

      const errorVisible = await page.locator('text=/memory|out of memory|too large/i').isVisible();
      expect(errorVisible).toBe(false);
    });

import { test, expect } from '@playwright/test';
import { FileHelper, ImageFactory, AudioFactory } from '../../fixtures';

test.describe('Progress Indicators', () => {
  test.describe.configure({ timeout: 120000 });

  test.beforeEach(async ({ page }) => {
    await page.goto('/convert');
    await page.waitForLoadState('networkidle');
  });

  test('shows progress bar during conversion', async ({ page }) => {
    // Use medium-sized file to ensure visible progress
    const imageBuffer = await ImageFactory.create({
      width: 2000,
      height: 2000,
      format: 'png'
    });

    const fileHelper = new FileHelper(page);
    await fileHelper.uploadBuffer(imageBuffer, 'progress-test.png', 'image/png');

    await expect(page.locator('.file-item')).toBeVisible();

    await page.locator('.format-selector').click();
    await page.locator('text=/JPEG/i').click();

    // Start conversion and immediately check for progress
    await page.locator('button:has-text("Convert")').click();

    // Progress bar should appear
    const progressBar = page.locator('.progress-bar, [role="progressbar"], .conversion-progress');
    await expect(progressBar).toBeVisible({ timeout: 5000 });

    // Wait for completion
    await expect(page.locator('.download-button, .success-indicator'))
      .toBeVisible({ timeout: 60000 });
  });

  test('progress updates continuously (not jumpy milestones)', async ({ page }) => {
    // Use larger file to see multiple progress updates
    const audioBuffer = await AudioFactory.create({
      duration: 30,  // 30 seconds of audio
      sampleRate: 44100,
      channels: 2
    });

    const fileHelper = new FileHelper(page);
    await fileHelper.uploadBuffer(audioBuffer, 'progress-test.wav', 'audio/wav');

    await expect(page.locator('.file-item')).toBeVisible();

    await page.locator('.format-selector').click();
    await page.locator('text=/MP3/i').click();

    // Capture progress values
    const progressValues: number[] = [];

    // Set up listener before starting conversion
    await page.exposeFunction('captureProgress', (value: number) => {
      progressValues.push(value);
    });

    await page.evaluate(() => {
      const observer = new MutationObserver((mutations) => {
        mutations.forEach((mutation) => {
          if (mutation.type === 'attributes' && mutation.attributeName === 'style') {
            const target = mutation.target as HTMLElement;
            const width = parseFloat(target.style.width || '0');
            if (width > 0) {
              (window as any).captureProgress(width);
            }
          }
        });
      });

      const progressBar = document.querySelector('.progress-bar, [role="progressbar"]');
      if (progressBar) {
        observer.observe(progressBar, { attributes: true });
      }
    });

    await page.locator('button:has-text("Convert")').click();

    await expect(page.locator('.download-button, .success-indicator'))
      .toBeVisible({ timeout: 120000 });

    // Verify multiple progress updates occurred
    console.log(`Progress values captured: ${progressValues.length}`);
    console.log(`Values: ${progressValues.join(', ')}`);

    // Should have more than 3-4 updates (not just milestone-based)
    // Per CONTEXT.md: "continuous updates, not jumpy milestones"
    expect(progressValues.length).toBeGreaterThan(3);
  });

  test('batch conversion shows overall progress', async ({ page }) => {
    // Upload multiple files
    const files = await Promise.all([
      ImageFactory.create({ width: 500, height: 500, format: 'png' }),
      ImageFactory.create({ width: 500, height: 500, format: 'png' }),
      ImageFactory.create({ width: 500, height: 500, format: 'png' })
    ]);

    const fileHelper = new FileHelper(page);
    for (let i = 0; i < files.length; i++) {
      await fileHelper.uploadBuffer(files[i], `batch-${i + 1}.png`, 'image/png');
    }

    await expect(page.locator('.file-item')).toHaveCount(3, { timeout: 10000 });

    // Select format and convert
    await page.locator('.format-selector').first().click();
    await page.locator('text=/JPEG/i').first().click();

    await page.locator('button:has-text("Convert")').click();

    // Look for batch progress indicator (e.g., "2/3 files")
    // Per CONTEXT.md: "Batch conversions show overall batch progress"
    const batchProgress = page.locator('text=/\\d+\\/\\d+.*file/i, .batch-progress');

    // This may or may not be implemented - document behavior
    const hasBatchProgress = await batchProgress.isVisible().catch(() => false);
    console.log(`Batch progress indicator visible: ${hasBatchProgress}`);

    // Wait for all to complete
    await expect(page.locator('.download-button, .success-indicator').first())
      .toBeVisible({ timeout: 60000 });
  });

  test.skip('cancel button stops conversion mid-process', async ({ page }) => {
    // Per CONTEXT.md: "Cancelable mid-process with cancel button that stops worker"
    // This requires cancel functionality to be implemented
    // Document as skipped until feature exists
  });

  test.skip('shows estimated time remaining', async ({ page }) => {
    // Per CONTEXT.md: "Show estimated time remaining"
    // This requires ETA feature to be implemented
    // Document as skipped until feature exists
  });
});
