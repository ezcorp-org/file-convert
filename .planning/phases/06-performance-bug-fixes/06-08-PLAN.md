```javascript
function audioBufferToWav(audioBuffer) {
  const numberOfChannels = audioBuffer.numberOfChannels;
  const sampleRate = audioBuffer.sampleRate;
  const format = 1; // PCM
  const bitDepth = 16;

  // Get audio data from all channels
  const length = audioBuffer.length * numberOfChannels;
  const samples = new Int16Array(length);

  for (let channel = 0; channel < numberOfChannels; channel++) {
    const channelData = audioBuffer.getChannelData(channel);
    for (let i = 0; i < audioBuffer.length; i++) {
      // Interleave channels and convert float32 to int16
      const sample = Math.max(-1, Math.min(1, channelData[i]));
      samples[i * numberOfChannels + channel] = sample < 0 ? sample * 0x8000 : sample * 0x7FFF;
    }
  }

  // Create WAV header + data
  const dataLength = samples.length * 2; // 2 bytes per sample (16-bit)
  const buffer = new ArrayBuffer(44 + dataLength);
  const view = new DataView(buffer);

  // RIFF header
  writeString(view, 0, 'RIFF');
  view.setUint32(4, 36 + dataLength, true);
  writeString(view, 8, 'WAVE');

  // fmt chunk
  writeString(view, 12, 'fmt ');
  view.setUint32(16, 16, true); // chunk size
  view.setUint16(20, format, true);
  view.setUint16(22, numberOfChannels, true);
  view.setUint32(24, sampleRate, true);
  view.setUint32(28, sampleRate * numberOfChannels * (bitDepth / 8), true);
  view.setUint16(32, numberOfChannels * (bitDepth / 8), true);
  view.setUint16(34, bitDepth, true);

  // data chunk
  writeString(view, 36, 'data');
  view.setUint32(40, dataLength, true);

  // Write samples
  const offset = 44;
  for (let i = 0; i < samples.length; i++) {
    view.setInt16(offset + i * 2, samples[i], true);
  }

  return buffer;
}

function writeString(view, offset, string) {
  for (let i = 0; i < string.length; i++) {
    view.setUint8(offset + i, string.charCodeAt(i));
  }
}
```

```typescript
/**
 * Decode audio file to WAV using worker thread
 * This prevents UI blocking during MP3/FLAC/OGG decoding
 */
private async decodeAudioToWAV(file: File, conversionId: string): Promise<File> {
  // Get the audio worker
  const worker = await this.workerManager.getWorker('audio');

  return new Promise<File>((resolve, reject) => {
    const messageHandler = (event: MessageEvent) => {
      if (event.data.id !== conversionId) return;

      switch (event.data.type) {
        case 'PROGRESS':
          // Update conversion progress
          this.updateProgress(conversionId, event.data.progress, event.data.message);
          break;

        case 'DECODE_RESULT':
          worker.removeEventListener('message', messageHandler);

          // Create File from the WAV buffer
          const originalName = file.name.substring(0, file.name.lastIndexOf('.')) || file.name;
          const wavBlob = new Blob([event.data.wavBuffer], { type: 'audio/wav' });
          const wavFile = new File([wavBlob], `${originalName}.wav`, { type: 'audio/wav' });

          resolve(wavFile);
          break;

        case 'ERROR':
          worker.removeEventListener('message', messageHandler);
          reject(new Error(event.data.error || 'Failed to decode audio'));
          break;
      }
    };

    worker.addEventListener('message', messageHandler);

    // Read file and send to worker
    file.arrayBuffer().then(arrayBuffer => {
      worker.postMessage({
        type: 'DECODE_AUDIO',
        id: conversionId,
        audioData: arrayBuffer,
        sampleRate: 44100, // Default, worker will detect from file
        channels: 2
      }, [arrayBuffer]); // Transfer for performance
    }).catch(reject);
  });
}
```
