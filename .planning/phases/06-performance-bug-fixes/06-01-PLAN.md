```typescript

  // ... check messageId
  // ... check progress id separately
  switch (event.data.type) {
    case 'RESULT':
      worker.removeEventListener('message', messageHandler);
      // ...
    case 'ERROR':
      worker.removeEventListener('message', messageHandler);
      // ...
  }
};
worker.addEventListener('message', messageHandler);
// ... postMessage
} catch (error) {
  this.handleConversionError(id, error as Error);
  // MESSAGE HANDLER NOT REMOVED!

```

After (fixed):
```typescript
const messageHandler = (event: MessageEvent) => {
  // Single authoritative ID check
  const msgId = event.data.id;
  if (msgId && msgId !== id) return;

  switch (event.data.type) {
    case 'RESULT':
      // Handle success (no removeEventListener - finally does it)
    case 'ERROR':
      // Handle error (no removeEventListener - finally does it)
  }
};

worker.addEventListener('message', messageHandler);
try {
  // ... postMessage and conversion logic
} catch (error) {
  this.handleConversionError(id, error as Error);
} finally {
  // ALWAYS remove handler
  worker.removeEventListener('message', messageHandler);
}
```
