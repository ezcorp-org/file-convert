/**
 * Validates text content as JSON
 * @returns true if valid JSON, false otherwise
 */
function validateJSON(content: string): boolean {
  try {
    JSON.parse(content);
    return true;
  } catch {
    return false;
  }
}

/**
 * Validates text content as CSV
 * Basic check: consistent column count across rows
 */
function validateCSV(content: string): boolean {
  const lines = content.trim().split('\n');
  if (lines.length === 0) return false;

  const expectedColumns = lines[0].split(',').length;
  return lines.every(line => {
    const columns = line.split(',').length;
    return columns === expectedColumns;
  });
}

/**
 * Validates text content as TSV
 * Basic check: consistent column count across rows
 */
function validateTSV(content: string): boolean {
  const lines = content.trim().split('\n');
  if (lines.length === 0) return false;

  const expectedColumns = lines[0].split('\t').length;
  return lines.every(line => {
    const columns = line.split('\t').length;
    return columns === expectedColumns;
  });
}

/**
 * Validates text content as YAML
 * Basic check: no obvious JSON-only syntax, proper YAML structure
 */
function validateYAML(content: string): boolean {
  // YAML is permissive - check it's not clearly invalid
  // Check for balanced quotes and valid structure
  const trimmed = content.trim();
  if (trimmed.length === 0) return false;

  // If it starts with { or [, it might be JSON not YAML
  // YAML can include JSON, but pure JSON files should fail
  // Check for YAML-specific features: colons with spaces, dashes for lists
  const hasYAMLStructure = /^[\w-]+:\s|^-\s/m.test(trimmed);
  const looksLikeJSON = /^[\[{]/.test(trimmed) && /[\]}]$/.test(trimmed);

  return hasYAMLStructure || !looksLikeJSON;
}
```

/**
 * Validates text file content based on expected format
 */
export async function validateTextFormat(file: File, format: string): Promise<boolean> {
  // Read file content as text
  const content = await file.text();

  switch (format.toLowerCase()) {
    case 'json':
      return validateJSON(content);
    case 'csv':
      return validateCSV(content);
    case 'tsv':
      return validateTSV(content);
    case 'yaml':
    case 'yml':
      return validateYAML(content);
    case 'txt':
    case 'md':
    case 'html':
      // These formats don't have strict structure
      return content.length > 0;
    default:
      return true; // Unknown formats pass by default
  }
}
```

import { describe, it, expect } from 'vitest';
import { validateTextFormat } from '$lib/utils/file-validation';

// Helper to create a mock File from content
function createTextFile(content: string, name: string, type = 'text/plain'): File {
  return new File([content], name, { type });
}

describe('Text Format Validation', () => {
  describe('JSON validation', () => {
    it('accepts valid JSON object', async () => {
      const file = createTextFile('{"name": "test"}', 'test.json');
      expect(await validateTextFormat(file, 'json')).toBe(true);
    });

    it('accepts valid JSON array', async () => {
      const file = createTextFile('[1, 2, 3]', 'test.json');
      expect(await validateTextFormat(file, 'json')).toBe(true);
    });

    it('rejects invalid JSON', async () => {
      const file = createTextFile('{name: "test"}', 'test.json'); // Missing quotes
      expect(await validateTextFormat(file, 'json')).toBe(false);
    });

    it('rejects empty content', async () => {
      const file = createTextFile('', 'test.json');
      expect(await validateTextFormat(file, 'json')).toBe(false);
    });
  });

  describe('CSV validation', () => {
    it('accepts valid CSV', async () => {
      const csv = 'name,age,city\nAlice,30,NYC\nBob,25,LA';
      const file = createTextFile(csv, 'test.csv');
      expect(await validateTextFormat(file, 'csv')).toBe(true);
    });

    it('rejects inconsistent columns', async () => {
      const csv = 'name,age,city\nAlice,30\nBob,25,LA,USA'; // Inconsistent
      const file = createTextFile(csv, 'test.csv');
      expect(await validateTextFormat(file, 'csv')).toBe(false);
    });
  });

  describe('TSV validation', () => {
    it('accepts valid TSV', async () => {
      const tsv = 'name\tage\tcity\nAlice\t30\tNYC';
      const file = createTextFile(tsv, 'test.tsv');
      expect(await validateTextFormat(file, 'tsv')).toBe(true);
    });
  });

  describe('YAML validation', () => {
    it('accepts valid YAML', async () => {
      const yaml = 'name: test\nage: 30';
      const file = createTextFile(yaml, 'test.yaml');
      expect(await validateTextFormat(file, 'yaml')).toBe(true);
    });

    it('accepts YAML list', async () => {
      const yaml = '- item1\n- item2\n- item3';
      const file = createTextFile(yaml, 'test.yaml');
      expect(await validateTextFormat(file, 'yaml')).toBe(true);
    });
