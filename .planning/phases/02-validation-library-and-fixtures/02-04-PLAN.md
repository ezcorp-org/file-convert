    3. Create DocumentFactory class with static methods:

       a) `createPDF(options?: DocumentFixtureOptions): Promise<Buffer>`
          - Use PDFKit to create PDF:
            ```typescript
            return new Promise((resolve, reject) => {
              const doc = new PDFDocument();
              const chunks: Buffer[] = [];

              doc.on('data', (chunk) => chunks.push(chunk));
              doc.on('end', () => resolve(Buffer.concat(chunks)));
              doc.on('error', reject);

              doc.fontSize(16).text(options.title ?? 'Test Document', { underline: true });
              doc.moveDown();
              doc.fontSize(12).text(options.content ?? 'This is test content.');

              // Add more paragraphs if requested
              for (let i = 1; i < (options.paragraphs ?? 1); i++) {
                doc.moveDown();
                doc.text(`Paragraph ${i + 1}: Lorem ipsum dolor sit amet.`);
              }

              doc.end();
            });
            ```

       b) `createTXT(options?: DocumentFixtureOptions): Buffer`
          - Generate plain text:
            ```typescript
            const title = options.title ?? 'Test Document';
            const content = options.content ?? 'This is test content.';
            let text = `${title}\n${'='.repeat(title.length)}\n\n${content}`;

            for (let i = 1; i < (options.paragraphs ?? 1); i++) {
              text += `\n\nParagraph ${i + 1}: Lorem ipsum dolor sit amet.`;
            }

            return Buffer.from(text, 'utf-8');
            ```

       c) `createHTML(options?: DocumentFixtureOptions): Buffer`

            ```typescript
            const content = options.content ?? 'This is test content.';
            let html = `<!DOCTYPE html>
            <html lang="en">
            <head><meta charset="UTF-8"><title>${title}</title></head>
            <body>
              <h1>${title}</h1>
              <p>${content}</p>`;

            for (let i = 1; i < (options.paragraphs ?? 1); i++) {
              html += `\n  <p>Paragraph ${i + 1}: Lorem ipsum dolor sit amet.</p>`;
            }

            html += '\n</body>\n</html>';
            return Buffer.from(html, 'utf-8');
            ```
